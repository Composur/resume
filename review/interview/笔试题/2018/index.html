1. 不考虑兼容性用javascript实现ajax 
        function ajax(methods,url) {
        var xhr = new XMLHttpRequest()
        xhr.open(methods, url, true)
        xhr.onreadystatechange = function () {
            if (xhr.readyState === 4 && xhr.status === 200) {
                return xhr.responseText
            }
        }
        xhr.send()
    }
    ajax('get','./data.json')
